# tools/backtester.py
import yfinance as yf
import pandas as pd

def run_backtest(portfolio: dict, period: str = "5y"):
    """
    portfolio: {"SCHD": 0.4, "JEPI": 0.3, "O": 0.2, "SHY": 0.1}
    period: "1y", "3y", "5y", "10y"
    """
    data = yf.download(list(portfolio.keys()), period=period)['Close']
    returns = data.pct_change().dropna()
    
    weighted_returns = sum(
        returns[ticker] * weight 
        for ticker, weight in portfolio.items()
    )
    
    cumulative = (1 + weighted_returns).cumprod()
    cagr = (cumulative.iloc[-1] ** (252 / len(cumulative))) - 1
    mdd = (cumulative / cumulative.cummax() - 1).min()
    
    return {
        "CAGR": f"{cagr:.2%}",
        "MDD": f"{mdd:.2%}",
        "Final Value": f"${cumulative.iloc[-1] * 10000:,.0f} (만달러 기준)"
    }
